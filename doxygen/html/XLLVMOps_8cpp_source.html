46 llvm::CallInst *createExternalLLVMIntrinsicCall(
47  llvm::IRBuilderBase &builder, LLVM::ModuleTranslation &moduleTranslation,
48  Operation *intrOp, llvm::StringRef intrinsicName) {
49  llvm::Type *resTy = nullptr;
50  unsigned numResults = intrOp->getNumResults();
51  if (numResults == 1)
52  resTy = moduleTranslation.convertType(*(intrOp->getResultTypes().begin()));
53  else if (numResults > 1) {
54  SmallVector<llvm::Type *> resTys;
55  for (auto ty : intrOp->getResultTypes())
56  resTys.push_back(moduleTranslation.convertType(ty));
57  resTy = llvm::StructType::get(builder.getContext(), resTys);
58  }
59  auto operands = moduleTranslation.lookupValues(intrOp->getOperands());
60  SmallVector<llvm::Type *> types;
61  for (auto op : operands)
62  types.push_back(op->getType());
63  llvm::Module *module = builder.GetInsertBlock()->getModule();
64  llvm::Function *llvmIntr =
65  getNamedIntrinsicDeclaration(module, intrinsicName, resTy, types);
66  return builder.CreateCall(llvmIntr, operands);
67 }
