27 struct DMAStartTaskOpPattern : OpConversionPattern<DMAStartTaskOp> {
28  using OpConversionPattern::OpConversionPattern;
29 
30  LogicalResult
31  matchAndRewrite(DMAStartTaskOp op, OpAdaptor adaptor,
32  ConversionPatternRewriter &rewriter) const override {
33  DMAConfigureTaskOp task_op = op.getTaskOp();
34  if (!task_op) {
35  // Cannot rewrite this; probably points to a DMAStartTaskForOp,
36  // which we will lower once it has been rewritten into a DMAStartTaskOp.
37  return failure();
38  }
39  AIE::TileOp tile = task_op.getTileOp();
40  std::optional<uint32_t> first_bd_id = task_op.getFirstBdId();
41  if (!first_bd_id) {
42  auto err = op.emitOpError(
43  "First buffer descriptor in chain has not been assigned an ID");
44  err.attachNote() << "Run the `aie-assign-runtime-buffer-descriptor-ids` "
45  "pass first or manually assign an ID.";
46  return failure();
47  }
48  rewriter.replaceOpWithNewOp<NpuPushQueueOp>(
49  op, tile.getCol(), tile.getRow(), task_op.getDirection(),
50  task_op.getChannel(), task_op.getIssueToken(), task_op.getRepeatCount(),
51  *first_bd_id);
52  return success();
53  }

83  bool shouldSkipBlock(Block &block) {
84  // Allow blocks in the input IR that contain nothing but a next_bd operation
85  // as the entry block. We will skip these blocks and not lower them to
86  // anything.

88  return block.isEntryBlock() && it.begin() == it.end();

91  LogicalResult verifyBdInBlock(Block &block) {
92  auto bd_ops = block.getOps<AIE::DMABDOp>();
93  // Exactly one BD op per block
94  int n_bd_ops = std::distance(bd_ops.begin(), bd_ops.end());
95  if (n_bd_ops < 1) {
96  auto error = block.getTerminator()->emitError(
97  "Block ending in this terminator does not contain a required "
98  "aie.dma_bd operation.");
99  error.attachNote(block.getParentOp()->getLoc())
100  << "Error encountered while lowering this BD configuration.";
101  return failure();
102  } else if (n_bd_ops > 1) {
103  auto error = block.getTerminator()->emitOpError(
104  "This block contains multiple aie.dma_bd operations. Exactly one is "
105  "required.");
106  auto it = bd_ops.begin();
107  ++it;
108  for (; it != bd_ops.end(); ++it) {
109  error.attachNote((*it)->getLoc()) << "Extra aie.dma_bd operation here.";
110  }
111  return failure();
112  }
113  AIE::DMABDOp bd_op = *bd_ops.begin();
114  if (!bd_op.getBdId().has_value()) {
115  auto error = bd_op.emitOpError(
116  "Cannot lower buffer descriptor without assigned ID.");
117  error.attachNote()
118  << "Run the `--aie-assign-runtime-sequence-bd-ids` pass first or "
119  "manually assign an ID to this buffer descriptor.";
120  error.attachNote(block.getParentOp()->getLoc())
121  << "Error encountered while lowering this BD configuration.";
122  return failure();
123  }
124  return success();
125  }

141  LogicalResult verifyNoUnsupportedOpsInBlock(Block &block) {
142  WalkResult unsupported_ops = block.walk([&](Operation *inner_op) {
143  return llvm::TypeSwitch<Operation *, WalkResult>(inner_op)
144  .Case<AIE::DMABDOp>(
145  [&](AIE::DMABDOp bd_op) { return WalkResult::advance(); })
146  .Case<AIE::UseLockOp>(
147  [&](AIE::UseLockOp lock_op) { return WalkResult::advance(); })
148  .Case<AIE::NextBDOp>(
149  [&](AIE::NextBDOp lock_op) { return WalkResult::advance(); })
150  .Case<AIE::EndOp>(
151  [&](AIE::EndOp lock_op) { return WalkResult::advance(); })
152  .Default([&](Operation *inner_op) {
153  auto error = block.getParentOp()->emitOpError(
154  "Unsupported operation within BD block.");
155  error.attachNote(inner_op->getLoc())
156  << "No lowering to NPU instructions available for this "
157  "operation.";
158  return WalkResult::interrupt();
159  });
160  });
161  if (unsupported_ops.wasInterrupted()) {
162  return failure();
163  }
164  return success();

180  LogicalResult setAddressForSingleBD(OpBuilder &builder, AIE::DMABDOp &bd_op,
181  AIE::TileOp &tile) {
182  uint32_t bd_id = bd_op.getBdId().value();
183  const AIE::AIETargetModel &target_model = AIE::getTargetModel(bd_op);
184  auto buf = bd_op.getBuffer();
185  uint64_t register_addr = getBufferDescriptorAddressRegisterAddress(
186  target_model, bd_id, tile.getCol(), tile.getRow());
187  if (mlir::BlockArgument buf_arg =
188  llvm::dyn_cast<mlir::BlockArgument>(buf)) {
189  if (!target_model.isShimNOCTile(tile.getCol(), tile.getRow())) {
190  return bd_op->emitOpError("DDR memory (runtime input arguments) can "
191  "only be referred to on shim tiles.");
192  }
193  unsigned arg_idx = buf_arg.getArgNumber();
194  int64_t offset = bd_op.getOffsetInBytes();
195  builder.create<NpuAddressPatchOp>(bd_op.getLoc(), /*addr*/ register_addr,
196  /*arg_idx*/ arg_idx,
197  /*arg_plus*/ offset);
198  } else if (AIE::BufferOp buffer =
199  llvm::dyn_cast<AIE::BufferOp>(buf.getDefiningOp())) {
200  uint64_t buf_addr;
201  if (!buffer.getAddress().has_value()) {
202  return bd_op->emitOpError(
203  "Cannot lower buffer without associated address. Run pass "
204  "--aie-assign-buffer-addresses first or manually assign an "
205  "address.");
206  }
207  buf_addr = *buffer.getAddress();
208  builder.create<NpuWrite32Op>(bd_op.getLoc(), register_addr, buf_addr,
209  nullptr, nullptr, nullptr);
210  } else {
211  return bd_op->emitOpError(
212  "Buffer argument must be either a constant aie.buffer or a runtime "
213  "sequence input argument.");
214  }
215  return success();
216  }

218  LogicalResult rewriteSingleBD(OpBuilder &builder, Block &block,
219  AIE::TileOp &tile,
220  AIE::DMAChannelDir channelDir) {
221  AIE::DMABDOp bd_op = getBdForBlock(block);
222  const auto &target_model = AIE::getTargetModel(bd_op);
223  MemRefType buffer_type = bd_op.getBuffer().getType();
224  uint32_t addr_granularity = target_model.getAddressGenGranularity();
225 
226  uint32_t bd_id = bd_op.getBdId().value();
227  int64_t offset = bd_op.getOffsetInBytes();
228  uint64_t len = bd_op.getLenInBytes();
229  uint64_t len_addr_granularity = len * 8 / addr_granularity;
230 
231  if (offset * 8 % addr_granularity != 0) {
232  return bd_op->emitOpError("Offset must be aligned to ")
233  << (addr_granularity / 8) << " byte boundary.";
234  }
235 
236  if (len < addr_granularity / 8) {
237  return bd_op->emitOpError("Transfer size of ")
238  << len << " bytes falls below minimum hardware transfer unit of "
239  << (addr_granularity / 8) << " bytes.";
240  }
241  // Process strides/wraps
242  std::optional<llvm::ArrayRef<AIE::BDDimLayoutAttr>> dims =
243  bd_op.getDimensions();
244  llvm::SmallVector<int64_t, 4> sizes = llvm::SmallVector<int64_t, 4>(4, 0);
245  llvm::SmallVector<int64_t, 4> strides = llvm::SmallVector<int64_t, 4>(4, 0);
246 
247  // Padding
248  std::optional<llvm::ArrayRef<AIE::BDPadLayoutAttr>> padDims =
249  bd_op.getPadDimensions();
250  llvm::SmallVector<int64_t, 4> padBefore =
251  llvm::SmallVector<int64_t, 4>(4, 0);
252  llvm::SmallVector<int64_t, 4> padAfter =
253  llvm::SmallVector<int64_t, 4>(4, 0);
254  std::fill(padBefore.begin(), padBefore.end(), 0);
255  std::fill(padAfter.begin(), padAfter.end(), 0);
256 
257  auto d0size = 0;
258  auto d0stride = 0;
259  auto d1size = 0;
260  auto d1stride = 0;
261  auto d2size = 0;
262  auto d2stride = 0;
263  auto iteration_size = 0;
264  auto iteration_stride = 0;
265 
266  if (dims && dims->size() > 0) {
267  llvm::SmallVector<int64_t, 4> input_sizes =
268  llvm::SmallVector<int64_t, 4>(4, 1);
269  llvm::SmallVector<int64_t, 4> input_strides =
270  llvm::SmallVector<int64_t, 4>(4, 0);
271  if (dims->size() > 4) {
272  return bd_op->emitOpError("At most four data layout transformation "
273  "dimensions may be provided.");
274  }
275 
276  for (size_t i = 0; i < dims->size(); i++) {
277  // Pass down dimensions in reverse order; in the MLIR, this allows
278  // us to specify step sizes/wraps in the same order as we would
279  // access a multi-dim C array, with the highest dimension first.
280  int j = dims->size() - i - 1;
281  input_sizes[i] = (*dims)[j].getSize();
282  input_strides[i] = (*dims)[j].getStride();
283  }
284 
285  // Do not check input_sizes[3] because a repeat can still be considered a
286  // linear transfer
287  bool isLinearTransfer = (input_sizes[0] >= 1) && (input_sizes[1] == 1) &&
288  (input_sizes[2] == 1);
289 
290  if (dims->size() > 2) {
291  d2size = (target_model.isMemTile(tile.getCol(), tile.getRow()))
292  ? (*dims)[2].getSize()
293  : 0;
294  }
295  if (padDims.has_value()) {
296  if (!target_model.isMemTile(tile.getCol(), tile.getRow()))
297  return bd_op->emitOpError()
298  << "Padding is only supported by memtile dma bds.";
299  if (padDims->size() > dims->size())
300  return bd_op->emitOpError()
301  << "Mismatch number of dimensions between padding(s)"
302  << " and wrap(s) and stride(s).";
303  if (channelDir == AIE::DMAChannelDir::MM2S) {
304  for (size_t i = 0; i < padDims->size(); i++) {
305  int j = padDims->size() - i - 1;
306  padBefore[i] = (*padDims)[j].getConstPadBefore();
307  padAfter[i] = (*padDims)[j].getConstPadAfter();
308  }
309  for (size_t i = padDims->size(); i < dims->size(); i++) {
310  padBefore[i] = 0;
311  padAfter[i] = 0;
312  }
313  } else
314  return bd_op->emitOpError()
315  << "supports padding only for MM2S direction on MemTiles.";
316  }
317  getHardwareStridesWraps(target_model, buffer_type, input_sizes,
318  input_strides, sizes, strides);
319 
320  if (failed(verifyStridesWraps(bd_op, buffer_type, tile.getCol(),
321  tile.getRow(), input_sizes, input_strides,
322  sizes, strides, isLinearTransfer))) {
323  return failure();
324  }
325 
326  iteration_size = sizes[3];
327  iteration_stride = strides[3];
328 
329  if (!isLinearTransfer) {
330  // d0_size, d0_stride
331  d0size = sizes[0];
332  d0stride = strides[0];
333 
334  // d1_size, d1_stride
335  d1size = sizes[1];
336  d1stride = strides[1];
337 
338  // d2_stride
339  d2stride = strides[2];
340  // d2_size set elsewhere
341  }
342  if (input_sizes[3] > 1 && input_strides[3] == 0) {
343  // We allow users to encode the repeat_count as a dimension 3 stride
344  // of 0. This must lower to a iteration wrap of 0, so no stride is
345  // ever added. We then repeat the BD using the repeat_count in
346  // NpuPushQueueOp.
347  iteration_size = 0;
348  iteration_stride = 0;
349  }
350 
351  // Ensure the total transfer length and the length expressed in the lowest
352  // three dimensions of strides/wraps agree. (Fourth dimension is
353  // iteration/repeat count and repeats the whole BD, so should not be
354  // incorporated in length of a single BD invocation.)
355  uint64_t len_dims_addr_granularity = 1;
356  for (size_t i = 0; i < 3; i++) {
357  len_dims_addr_granularity *= sizes[i];
358  }
359  if (len_dims_addr_granularity != len_addr_granularity) {
360  auto err =
361  bd_op->emitOpError(
362  "Buffer descriptor length does not match length of transfer "
363  "expressed by lowest three dimensions of data layout "
364  "transformation strides/wraps. ")
365  << "BD length is " << (len_addr_granularity * addr_granularity / 8)
366  << " bytes. "
367  << "Lowest three dimensions of data layout transformation would "
368  "result in transfer of "
369  << (len_dims_addr_granularity * addr_granularity / 8) << " bytes. ";
370  err.attachNote() << "Do not include the highest dimension size in "
371  "transfer length, as this is the BD repeat count.";
372  return failure();
373  }
374  } else {
375  if (padDims && target_model.isMemTile(tile.getCol(), tile.getRow()) &&
376  channelDir == AIE::DMAChannelDir::MM2S) {
377  return bd_op->emitOpError()
378  << "Padding requires n-d data layouts expressed as "
379  << "wrap(s) and stride(s).";
380  } else if (padDims) {
381  return bd_op->emitOpError() << "Padding is supported only on MemTiles.";
382  }
383  }
384  // find next BD ID, if any
385  uint32_t use_next_bd = 0;
386  uint32_t next_bd_id = 0;
387  if (bd_op.getNextBdId().has_value()) {
388  next_bd_id = bd_op.getNextBdId().value();
389  use_next_bd = 1;
390  }
391 
392  builder.create<NpuWriteBdOp>(
393  bd_op.getLoc(), tile.getCol(), bd_id, len_addr_granularity, offset, 0,
394  0, 0, 0,
395  /* TODO: Strides/Wraps */
396  /*d0_size=*/d0size, /*d0_stride=*/d0stride,
397  /*d1_size=*/d1size, /*d1_stride=*/d1stride,
398  /*d2_size=*/d2size, /*d2_stride=*/d2stride,
399  /*iteration_current=*/0, /*iteration_size=*/iteration_size,
400  /*iteration_stride=*/iteration_stride,
401  /* TODO: Next BD */
402  /*next_bd=*/next_bd_id,
403  /*row=*/tile.getRow(),
404  /*use_next_bd=*/use_next_bd,
405  /*valid_bd=*/1,
406  /* TODO: Locks */
407  /*lock_rel_val=*/0, /*lock_rel_id=*/0, /*lock_acq_enable=*/0,
408  /*lock_acq_val=*/0, /*lock_ackq_id=*/0, /*d0_zero_before=*/padBefore[0],
409  /*d1_zero_before=*/padBefore[1], /*d2_zero_before=*/padBefore[2],
410  /*d0_zero_after=*/padAfter[0], /*d1_zero_after=*/padAfter[1],
411  /*d2_zero_after=*/padAfter[2]);
412  return setAddressForSingleBD(builder, bd_op, tile);
413  }

415  LogicalResult hoistNextBdOpsIntoAttrs(DMAConfigureTaskOp op) {
416  Region &body = op.getBody();
417  for (auto it = body.begin(); it != body.end(); ++it) {
418  Block &block = *it;
419  if (shouldSkipBlock(block)) {
420  continue;
421  }
422  AIE::DMABDOp bd_op = getBdForBlock(block);
423  if (AIE::NextBDOp next_bd_op =
424  llvm::dyn_cast<AIE::NextBDOp>(block.getTerminator())) {
425  if (bd_op.getNextBdId().has_value()) {
426  auto error =
427  bd_op.emitOpError("Cannot specify both next_bd_id attribute and "
428  "aie.next_bd operation.");
429  error.attachNote(next_bd_op.getLoc())
430  << "Potentially conflicting next buffer descriptor ID specified "
431  "here.";
432  return failure();
433  }
434  Block &next_bd_block = *next_bd_op.getDest();
435  AIE::DMABDOp next_dma_bd_op = getBdForBlock(next_bd_block);
436  assert(next_dma_bd_op.getBdId()
437  .has_value()); // Next BD should have assigned ID, and this
438  // should have been checked by earlier
439  // verifyBdInBlock() call
440  bd_op.setNextBdId(next_dma_bd_op.getBdId().value());
441  OpBuilder builder(next_bd_op);
442  builder.create<AIE::EndOp>(next_bd_op.getLoc());
443  next_bd_op.erase();
444  }
445  }
446  return success();
447  }

449  LogicalResult rewriteSingleDMAConfigureTaskOp(DMAConfigureTaskOp op) {
450  OpBuilder builder(op);
451  AIE::TileOp tile = op.getTileOp();
452 
453  if (!op.use_empty()) {
454  auto err = op.emitOpError("Cannot lower while op still has uses.");
455  mlir::Operation::use_range uses = op.getOperation()->getUses();
456  for (auto it = uses.begin(); it != uses.end(); ++it) {
457  err.attachNote(it->getOwner()->getLoc()) << "Used here.";
458  }
459  return failure();
460  }
461 
462  Region &body = op.getBody();
463 
464  // Verify each BD block first; subsequent functions rely on them being
465  // well-formed
466  for (auto it = body.begin(); it != body.end(); ++it) {
467  if (shouldSkipBlock(*it)) {
468  continue;
469  }
470  if (failed(verifyNoUnsupportedOpsInBlock(*it))) {
471  return failure();
472  }
473  if (failed(verifyBdInBlock(*it))) {
474  return failure();
475  }
476  if (failed(verifyOptionalLocksInBlock(*it))) {
477  return failure();
478  }
479  }
480 
481  // Hoist next_bd operations into next_bd_id attribute of the dma_bd
482  if (failed(hoistNextBdOpsIntoAttrs(op))) {
483  return failure();
484  }
485 
486  auto channelDir = op.getDirection();
487 
488  // Lower all BDs
489  for (auto it = body.begin(); it != body.end(); ++it) {
490  Block &block = *it;
491  if (shouldSkipBlock(block)) {
492  continue;
493  }
494  if (failed(rewriteSingleBD(builder, block, tile, channelDir))) {
495  return failure();
496  }
497  }
498 
499  op.erase();
500 
501  return success();
502  }

504  LogicalResult rewriteDMAConfigureTaskOp(AIE::DeviceOp device) {

517  void runOnOperation() override {
518  AIE::DeviceOp device = getOperation();
519 
520  // Convert DMAStartBD and DMAAwaitBD ops
521  ConversionTarget target(getContext());
522  target.addLegalDialect<AIEXDialect>();
523  target.addIllegalOp<DMAStartTaskOp>();
524  target.addIllegalOp<DMAAwaitTaskOp>();
525  RewritePatternSet patterns(&getContext());
526  patterns.insert<DMAStartTaskOpPattern>(&getContext());
527  patterns.insert<DMAAwaitTaskOpPattern>(&getContext());
528  if (failed(applyPartialConversion(device, target, std::move(patterns)))) {
529  signalPassFailure();
530  }
531 
532  // Lower the configuration for the BDs
533  if (failed(rewriteDMAConfigureTaskOp(device))) {
534  signalPassFailure();
535  }
536  }
