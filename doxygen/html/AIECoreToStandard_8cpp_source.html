50 static auto getAIE1Intrinsics(OpBuilder &builder) {
51  Type int32Type = IntegerType::get(builder.getContext(), 32);
52  Type int128Type = IntegerType::get(builder.getContext(), 128);
53  Type int384Type = IntegerType::get(builder.getContext(), 384);
54  Type floatType = FloatType::getF32(builder.getContext());
55 
56  // Note that not all of these are valid for a particular design, or needed.
57  // For right now, we will just accept the noise.
58  IntrinsicDecls functions = {
59  {"debug_i32", {int32Type}, {}},
60  {"llvm.aie.event0", {}, {}},
61  {"llvm.aie.event1", {}, {}},
62  {"llvm.aie.put.ms",
63  {int32Type, int32Type},
64  {}}, //(%channel, %value) -> ()
65  {"llvm.aie.put.wms",
66  {int32Type, int128Type},
67  {}}, //(%channel, %value) -> ()
68  {"llvm.aie.put.fms",
69  {int32Type, floatType},
70  {}}, //(%channel, %value) -> ()
71  {"llvm.aie.get.ss", {int32Type}, {int32Type}}, //(%channel, %value) -> ()
72  {"llvm.aie.get.wss",
73  {int32Type},
74  {int128Type}}, //(%channel, %value) -> ()
75  {"llvm.aie.get.fss", {int32Type}, {floatType}}, //(%channel, %value) -> ()
76  {"llvm.aie.put.mcd", {int384Type}, {}},
77  {"llvm.aie.get.scd", {}, {int384Type}},
78  {"llvm.aie.lock.acquire.reg",
79  {int32Type, int32Type},
80  {}}, //(%lock_id, %lock_val) -> ()
81  {"llvm.aie.lock.release.reg",
82  {int32Type, int32Type},
83  {}}, //(%lock_id, %lock_val) -> ()
84  };
85  return functions;
86 }

111 static void declareAIEIntrinsics(AIEArch arch, OpBuilder &builder) {
112  auto registerIntrinsics = [&builder](IntrinsicDecls functions) {
113  for (auto &i : functions) {
114  auto [name, argTypes, retTypes] = i;
115  builder
116  .create<func::FuncOp>(
117  builder.getUnknownLoc(), name,
118  FunctionType::get(builder.getContext(), argTypes, retTypes))
119  .setPrivate();
120  }
121  };
122  switch (arch) {
123  case AIEArch::AIE1:
124  registerIntrinsics(getAIE1Intrinsics(builder));
125  return;
126  case AIEArch::AIE2:
127  case AIEArch::AIE2p:
128  registerIntrinsics(getAIE2Intrinsics(builder));
129  return;
130  }
131  llvm::report_fatal_error("unsupported arch");
132 }

332 struct AIEUseLockToStdLowering : OpConversionPattern<UseLockOp> {
333  using OpConversionPattern::OpConversionPattern;
334  ModuleOp &module;
335 
336  AIEUseLockToStdLowering(MLIRContext *context, ModuleOp &m,
337  PatternBenefit benefit = 1)
338  : OpConversionPattern(context, benefit), module(m) {}
339  LogicalResult
340  matchAndRewrite(UseLockOp useLock, OpAdaptor adaptor,
341  ConversionPatternRewriter &rewriter) const override {
342  if (!isa<DeviceOp>(useLock->getParentOp())) {
343  auto device = useLock->getParentOfType<DeviceOp>();
344  if (!device) {
345  return module.emitOpError("Device Not found!");
346  }
347  const auto &targetModel = device.getTargetModel();
348 
349  // Generate the intrinsic name
350  std::string funcName;
351  if (targetModel.getTargetArch() == AIEArch::AIE1)
352  funcName = "llvm.aie.lock.";
353  else
354  funcName = "llvm.aie2.";
355  if (useLock.acquire() || useLock.acquireGE())
356  funcName += "acquire";
357  else if (useLock.release())
358  funcName += "release";
359  if (targetModel.getTargetArch() == AIEArch::AIE1)
360  funcName += ".reg";
361 
362  auto useLockFunc = module.lookupSymbol<func::FuncOp>(funcName);
363  if (!useLockFunc)
364  return useLock.emitOpError("Could not find the intrinsic function!");
365 
366  SmallVector<Value, 2> args;
367  auto lockValue = useLock.getLockValue();
368 
369  // AIE2 acquire greater equal is encoded as a negative value.
370  if (useLock.acquireGE()) {
371  lockValue = -lockValue;
372  }
373  args.push_back(rewriter.create<arith::IndexCastOp>(
374  useLock.getLoc(), IntegerType::get(rewriter.getContext(), 32),
375  useLock.getLock()));
376  args.push_back(rewriter.create<arith::ConstantOp>(
377  useLock.getLoc(), IntegerType::get(rewriter.getContext(), 32),
378  rewriter.getI32IntegerAttr(lockValue)));
379 
380  rewriter.create<func::CallOp>(rewriter.getUnknownLoc(), useLockFunc,
381  args);
382  }
383  rewriter.eraseOp(useLock);
384  return success();
385  }

388 struct AIEBufferToStandard : OpConversionPattern<BufferOp> {
389  using OpConversionPattern::OpConversionPattern;
390  ModuleOp &module;
391  int tileCol = 0;
392  int tileRow = 0;
393  AIEBufferToStandard(MLIRContext *context, ModuleOp &m,
394  PatternBenefit benefit = 1, int tileCol = -1,
395  int tileRow = -1)
396  : OpConversionPattern(context, benefit), module(m), tileCol(tileCol),
397  tileRow(tileRow) {}
398  LogicalResult
399  matchAndRewrite(BufferOp buffer, OpAdaptor adaptor,
400  ConversionPatternRewriter &rewriter) const override {
401  rewriter.setInsertionPointToStart(module.getBody());
402  auto t = llvm::cast<MemRefType>(buffer.getType());
403  int col = llvm::cast<TileOp>(buffer.getTile().getDefiningOp()).getCol();
404  int row = llvm::cast<TileOp>(buffer.getTile().getDefiningOp()).getRow();
405  auto symName = buffer.name().getValue();
406  mlir::ElementsAttr initValue = buffer.getInitialValueAttr();
407  // Don't emit initialization for cores that don't "own" the buffer (to
408  // prevent duplication in the data section of the elf/object file)
409  if ((tileRow != row && tileRow != -1) || (tileCol != col && tileCol != -1))
410  initValue = nullptr;
411  rewriter.create<memref::GlobalOp>(
412  rewriter.getUnknownLoc(), symName, rewriter.getStringAttr("public"),
413  buffer.getType(), initValue, /*constant*/ false,
414  /*alignment*/ nullptr);
415 
416  for (auto &use : make_early_inc_range(buffer.getResult().getUses())) {
417  Operation *user = use.getOwner();
418  rewriter.setInsertionPoint(user);
419  auto allocated = rewriter.create<memref::GetGlobalOp>(
420  rewriter.getUnknownLoc(), t, symName);
421  // Assume that buffers are aligned so they can be vectorized.
422  rewriter.create<memref::AssumeAlignmentOp>(rewriter.getUnknownLoc(),
423  allocated, 32);
424 
425  use.set(allocated.getResult());
426  }
427 
428  rewriter.eraseOp(buffer);
429  return success();
430  }
431 };

433 struct AIECoreToStandardFunc : OpConversionPattern<CoreOp> {
434  using OpConversionPattern::OpConversionPattern;
435  ModuleOp &module;
436  IRMapping &mapper;
437  DenseMap<Operation *, SmallVector<BufferOp, 4>> &tileToBuffers;
438  int tileCol = 0;
439  int tileRow = 0;
440 
441  AIECoreToStandardFunc(
442  MLIRContext *context, ModuleOp &m, IRMapping &mapper,
443  DenseMap<Operation *, SmallVector<BufferOp, 4>> &tileToBuffers,
444  PatternBenefit benefit = 1, int tileCol = 1, int tileRow = 1)
445  : OpConversionPattern(context, benefit), module(m), mapper(mapper),
446  tileToBuffers(tileToBuffers), tileCol(tileCol), tileRow(tileRow) {}
447 
448  LogicalResult
449  matchAndRewrite(CoreOp op, OpAdaptor adaptor,
450  ConversionPatternRewriter &rewriter) const override {
451 
452  int col = op.colIndex();
453  int row = op.rowIndex();
454 
455  // Only pull code for the indicated function
456  if ((tileRow != row && tileRow != -1) ||
457  (tileCol != col && tileCol != -1)) {
458  rewriter.eraseOp(op);
459  return success();
460  }
461 
462  // The parent should be an AIE.device op.
463  rewriter.setInsertionPointAfter(op->getParentOp());
464 
465  std::string coreName("core_" + std::to_string(col) + "_" +
466  std::to_string(row));
467  auto coreFunc = rewriter.create<func::FuncOp>(
468  rewriter.getUnknownLoc(), coreName,
469  FunctionType::get(rewriter.getContext(), {}, {}));
470 
471  rewriter.cloneRegionBefore(op.getBody(), coreFunc.getBody(),
472  coreFunc.getBody().begin(), mapper);
473 
474  // Rewrite the AIE.end() op
475  coreFunc.getBody().walk([&](Operation *childOp) {
476  rewriter.setInsertionPointAfter(childOp);
477 
478  if (isa<EndOp>(childOp)) {
479  rewriter.create<func::ReturnOp>(rewriter.getUnknownLoc(),
480  ValueRange({}));
481  rewriter.eraseOp(childOp);
482  }
483  });
484 
485  rewriter.eraseOp(op);
486  return success();
487  }
488 };

525 struct AIECoreToStandardPass : AIECoreToStandardBase<AIECoreToStandardPass> {
526  void runOnOperation() override {
527 
528  ModuleOp m = getOperation();
529  OpBuilder builder = OpBuilder::atBlockEnd(m.getBody());
530 
531  if (m.getOps<DeviceOp>().empty()) {
532  m.emitOpError("expected AIE.device operation at toplevel");
533  return signalPassFailure();
534  }
535  DeviceOp device = *m.getOps<DeviceOp>().begin();
536  const auto &targetModel = device.getTargetModel();
537 
538  // Ensure that we don't have an incorrect target triple. This may override
539  // some bogus target triple in the original mlir.
540  m->setAttr(LLVM::LLVMDialect::getTargetTripleAttrName(),
541  builder.getStringAttr(
542  getArchIntrinsicString(targetModel.getTargetArch())));
543 
544  DenseMap<Operation *, SmallVector<BufferOp, 4>> tileToBuffers;
545 
546  // Populate intrinsic functions
547  // Intrinsic information:
548  // peano/llvm-project/llvm/lib/Target/AIE/AIEInstrInfo.td Also take a look
549  // at the tests: peano/llvm-project/llvm/test/CodeGen/AIE
550  builder.setInsertionPointToStart(m.getBody());
551  declareAIEIntrinsics(targetModel.getTargetArch(), builder);
552 
553  IRMapping mapper;
554  ConversionTarget target(getContext());
555  target.addLegalDialect<func::FuncDialect>();
556  target.addLegalDialect<cf::ControlFlowDialect>();
557  target.addLegalDialect<memref::MemRefDialect>();
558  target.addLegalDialect<VectorDialect>();
559  target.addLegalDialect<arith::ArithDialect>();
560  target.addLegalDialect<math::MathDialect>();
561  target.addLegalDialect<index::IndexDialect>();
562  target.addLegalOp<func::FuncOp, ModuleOp>();
563 
564  RewritePatternSet patterns(&getContext());
565  patterns.add<AIEPutStreamToStdLowering, AIEGetStreamToStdLowering,
566  AIEPutCascadeToStdLowering, AIEGetCascadeToStdLowering,
567  AIEDebugOpToStdLowering, AIEUseLockToStdLowering,
568  AIEEventOpToStdLowering>(m.getContext(), m);
569 
570  patterns.add<AIEBufferToStandard>(m.getContext(), m, /*benefit*/ 1, tileCol,
571  tileRow);
572  if (failed(applyPartialConversion(m, target, std::move(patterns))))
573  return signalPassFailure();
574 
575  RewritePatternSet outlinePatterns(&getContext());
576  outlinePatterns.add<AIECoreToStandardFunc>(m.getContext(), m, mapper,
577  tileToBuffers, /*benefit*/ 1,
578  tileCol, tileRow);
579  if (failed(applyPartialConversion(m, target, std::move(outlinePatterns))))
580  return signalPassFailure();
581 
582  // Move all the func.func ops and memref.globals from the device to the
583  // module
584  outlineOps<memref::GlobalOp>(device);
585  outlineOps<func::FuncOp>(device);
586 
587  RewritePatternSet removepatterns(&getContext());
