44 class LockAnalysis {
45  DenseMap<std::pair<Value, int>, int> locksPerTile;
46 
47 public:
48  LockAnalysis(DeviceOp &device) {
49  // go over the locks created for each tile and update the index in
50  // locksPerTile
51  device.walk([&](LockOp lockOp) {
52  auto tile = lockOp.getTile();
53  auto lockID = lockOp.getLockIDValue();
54  locksPerTile[{tile, lockID}] = 1;
55  });
56  }
57 
58  /// Given a tile, returns next usable lockID for that tile.
59  int getLockID(TileOp &tileOp) {
60  const auto &targetModel = getTargetModel(tileOp);
61  for (unsigned i = 0;
62  i < targetModel.getNumLocks(tileOp.getCol(), tileOp.getRow()); i++)
63  if (int usageCnt = locksPerTile[{tileOp, i}]; usageCnt == 0) {
64  locksPerTile[{tileOp, i}] = 1;
65  return i;
66  }
67  return -1;
68  }
69 };

109  /// Given a tile and DMAChannelDir, returns next usable channel index for
110  /// that tile.
111  int getDMAChannelIndex(TileOp tileOp, DMAChannelDir dir) {
112  const auto &targetModel = getTargetModel(tileOp);
113  int maxChannelNum = 0;
114  if (tileOp.isShimTile())
115  maxChannelNum = 2;
116  else {
117  if (dir == DMAChannelDir::MM2S)
118  maxChannelNum = targetModel.getNumSourceSwitchboxConnections(
119  tileOp.getCol(), tileOp.getRow(), WireBundle::DMA);
120  else
121  maxChannelNum = targetModel.getNumDestSwitchboxConnections(
122  tileOp.getCol(), tileOp.getRow(), WireBundle::DMA);
123  }
124  for (int i = 0; i < maxChannelNum; i++)
125  if (int usageCnt = channelsPerTile[{tileOp.getResult(), dir, i}];
126  usageCnt == 0) {
127  channelsPerTile[{tileOp.getResult(), dir, i}] = 1;
128  return i;
129  }
130  return -1;
131  }

156  /// Function that returns true if two tiles in the AIE array share a memory
157  /// module. share_direction is equal to:
158  /// * -1 if the shared memory module is that of the first input tile,
159  /// * 1 if it is that of the second input tile,
160  /// * 0 is no memory module is shared.
161  bool isSharedMemory(TileOp a, TileOp b, int *share_direction) {
162  const auto &targetModel = getTargetModel(a.getOperation());
163 
164  if ((a.isShimTile() && !b.isShimTile()) ||
165  (!a.isShimTile() && b.isShimTile())) {
166  *share_direction = 0;
167  return false;
168  }
169  if ((targetModel.isMemTile(a.getCol(), a.getRow()) &&
170  !targetModel.isMemTile(b.getCol(), b.getRow())) ||
171  (!targetModel.isMemTile(a.getCol(), a.getRow()) &&
172  targetModel.isMemTile(b.getCol(), b.getRow()))) {
173  *share_direction = 0;
174  return false;
175  }
176  bool rightShared = targetModel.isLegalMemAffinity(
177  a.colIndex(), a.rowIndex(), b.colIndex(), b.rowIndex());
178 
179  bool leftShared = targetModel.isLegalMemAffinity(
180  b.colIndex(), b.rowIndex(), a.colIndex(), a.rowIndex());
181 
182  if (leftShared)
183  *share_direction = -1;
184  else if (rightShared)
185  *share_direction = 1;
186  else
187  *share_direction = 0;
188 
189  return leftShared || rightShared;
190  }

192  // Return true if the objectFifo created by createOp requires a DMA to be set
193  // up. This is the case if the tiles are not adjacent (no shared memory), if
194  // the objectFifo broadcasts to multiple tiles, if one of the consumers or
195  // the producer wants to use the multi-dimensional address generation
196  // features of the DMA, if the objectFifo is part of a LinkOp, or if the
197  // via_DMA or repeatCount attributes of the objectFifo are set.
198  bool requiresDMAs(ObjectFifoCreateOp createOp, int &share_direction) {
199  bool hasSharedMemory = false;
200  bool atLeastOneConsumerWantsTransform = false;
201  bool isUsedInLinkOp = false;
202 
203  if (createOp.getVia_DMA())
204  return true;
205 
206  if (createOp.getRepeatCount().has_value())
207  return true;
208 
209  if (createOp.getConsumerTiles().size() == 1 &&
210  createOp.getDimensionsToStream().empty()) {
211 
212  // Test for shared memory
213  for (auto consumerTile : createOp.getConsumerTiles()) {
214  if (auto consumerTileOp =
215  dyn_cast<TileOp>(consumerTile.getDefiningOp())) {
216  if (std::count(splitBecauseLink.begin(), splitBecauseLink.end(),
217  createOp))
218  hasSharedMemory =
219  isSharedMemory(createOp.getProducerTileOp(),
220  createOp.getProducerTileOp(), &share_direction);
221  else
222  hasSharedMemory = isSharedMemory(createOp.getProducerTileOp(),
223  consumerTileOp, &share_direction);
224  }
225  }
226  }
227 
228  // Only test for use of data layout transformations if we are in the shared
229  // memory case; otherwise, we will return `true` in any case.
230  if (hasSharedMemory) {
231  // Even if just one of the consumers in the list of consumers wants to
232  // perform a memory transform, we need to use DMAs.
233  for (BDDimLayoutArrayAttr dims :
234  createOp.getDimensionsFromStreamPerConsumer())
235  if (!dims.empty()) {
236  atLeastOneConsumerWantsTransform = true;
237  break;
238  }
239  }
240 
241  // Check if the objectfifo operation can use shared memory for linking. If
242  // the link operation is a distribute or a join operation, or if the link
243  // has different memref types, DMAs are required even if shared memory is
244  // available and the objectfifo should be split. Otherwise also check if the
245  // via_shared_memory attribute of the objectfifo operation is set and try to
246  // apply it.
247  if (hasSharedMemory) {
248  if (auto linkOp = getOptionalLinkOp(createOp)) {
249  isUsedInLinkOp = true;
250  int share_dir = 0;
251  if (!linkOp->isDistribute() && !linkOp->isJoin()) {
252  auto fifoInType = llvm::cast<AIEObjectFifoType>(
253  linkOp->getInputObjectFifos()[0].getElemType());
254  auto producerType =
255  llvm::cast<MemRefType>(fifoInType.getElementType());
256  auto fifoOutType = llvm::cast<AIEObjectFifoType>(
257  linkOp->getOutputObjectFifos()[0].getElemType());
258  auto consumerType =
259  llvm::cast<MemRefType>(fifoOutType.getElementType());
260  if (consumerType != producerType) {
261  // TODO: Support for different memref types through shared
262  // memory without DMAs
263  splitBecauseLink.push_back(createOp);
264  }
265  if (createOp.getViaSharedMem().has_value()) {
266  checkAndApplyViaSharedMemAttribute(createOp, share_dir);
267  if (share_direction == share_dir)
268  isUsedInLinkOp = false;
269  else
270  splitBecauseLink.push_back(createOp);
271  }
272  } else {
273  splitBecauseLink.push_back(createOp);
274  }
275  }
276  }
277 
278  return !hasSharedMemory || atLeastOneConsumerWantsTransform ||
279  isUsedInLinkOp;
280  }

282  // Checks if via_shared_mem attribute of the objectfifo is set and if so
283  // tries to apply it. If the desired shared memory module is available to
284  // both producer and consumer then it will be used, otherwise a warning is
285  // emitted and the original shared memory module is used instead.
286  void checkAndApplyViaSharedMemAttribute(ObjectFifoCreateOp createOp,
287  int &share_direction) {
288  if (createOp.getViaSharedMem().has_value()) {
289  int desiredSharedTile = createOp.getViaSharedMem().value();
290  int desiredSharedModule = 1;
291  if (desiredSharedTile == 0)
292  desiredSharedModule = -1;
293  if (share_direction != desiredSharedModule) {
294  bool desiredSharedModuleIsShared = false;
295  int newShareDirection = 0;
296  for (auto consumerTile : createOp.getConsumerTiles()) {
297  if (auto consumerTileOp =
298  dyn_cast<TileOp>(consumerTile.getDefiningOp()))
299  if (share_direction == -1)
300  /// * -1 if the shared memory module is that of the first input
301  /// tile,
302  /// * 1 if it is that of the second input tile
303  desiredSharedModuleIsShared =
304  isSharedMemory(consumerTileOp, createOp.getProducerTileOp(),
305  &newShareDirection);
306  }
307  if (desiredSharedModuleIsShared) {
308  if (share_direction == newShareDirection)
309  share_direction = (share_direction == -1) ? 1 : -1;
310  else
311  createOp->emitWarning("Memory module specified by `via_shared_mem` "
312  "is not available as shared memory module");
313  }
314  }
315  }

345  /// Function used to create objectFifo locks based on target architecture.
346  /// Called by createObjectFifoElements().
347  std::vector<LockOp> createObjectFifoLocks(OpBuilder &builder,
348  LockAnalysis &lockAnalysis,
349  ObjectFifoCreateOp op, int numElem,
350  TileOp creation_tile,
351  int repeatCount) {
352  std::vector<LockOp> locks;
353  if (op.getDisableSynchronization())
354  return locks;
355  auto dev = op->getParentOfType<DeviceOp>();
356  auto &target = dev.getTargetModel();
357  // if shimTile external buffers are collected from input code
358  // create as many locks as there are external buffers
359  if (creation_tile.isShimTile()) {
360  numElem = 1;
361  if (!externalBuffersPerFifo[op].empty())
362  numElem = externalBuffersPerFifo[op].size();
363  }
364  if (target.getTargetArch() == AIEArch::AIE1) {
365  int of_elem_index = 0; // used to give objectFifo elements a symbolic name
366  for (int i = 0; i < numElem; i++) {
367  // create corresponding aie1 locks
368  int initValue = op.getInitValues().has_value() ? 1 : 0;
369  int lockID = lockAnalysis.getLockID(creation_tile);
370  assert(lockID >= 0 && "No more locks to allocate!");
371  auto lock = builder.create<LockOp>(builder.getUnknownLoc(),
372  creation_tile, lockID, initValue);
373  lock.getOperation()->setAttr(
374  SymbolTable::getSymbolAttrName(),
375  builder.getStringAttr(op.name().str() + "_lock_" +
376  std::to_string(of_elem_index)));
377  locks.push_back(lock);
378  of_elem_index++;
379  }
380  } else {
381  // create corresponding aie2 locks
382  auto initValues = op.getInitValues().has_value()
383  ? op.getInitValues().value().size()
384  : 0;
385  int prodLockID = lockAnalysis.getLockID(creation_tile);
386  assert(prodLockID >= 0 && "No more locks to allocate!");
387  int prodLockValue = (numElem - initValues) * repeatCount;
388  auto prodLock = builder.create<LockOp>(
389  builder.getUnknownLoc(), creation_tile, prodLockID, prodLockValue);
390  prodLock.getOperation()->setAttr(
391  SymbolTable::getSymbolAttrName(),
392  builder.getStringAttr(op.name().str() + "_prod_lock"));
393  locks.push_back(prodLock);
394 
395  int consLockID = lockAnalysis.getLockID(creation_tile);
396  assert(consLockID >= 0 && "No more locks to allocate!");
397  int consLockValue = initValues * repeatCount;
398  auto consLock = builder.create<LockOp>(
399  builder.getUnknownLoc(), creation_tile, consLockID, consLockValue);
400  consLock.getOperation()->setAttr(
401  SymbolTable::getSymbolAttrName(),
402  builder.getStringAttr(op.name().str() + "_cons_lock"));
403  locks.push_back(consLock);
404  }
405  return locks;
406  }

408  /// Function used to create objectFifo elements and their locks.
409  /// It maps the input objectFifo to associated buffers and locks.
410  void createObjectFifoElements(OpBuilder &builder, LockAnalysis &lockAnalysis,
411  ObjectFifoCreateOp op, int share_direction) {
412  if (!op.size())
413  return;
414 
415  std::vector<BufferOp> buffers;
416  auto fifo = llvm::cast<AIEObjectFifoType>(op.getElemType());
417  auto elemType = llvm::cast<MemRefType>(fifo.getElementType());
418  int numElem = op.size();
419  int of_elem_index = 0; // used to give objectFifo elements a symbolic name
420 
421  // if this objectFifo is linked to another, check if the other's elements
422  // have already been created: if none of the output objectfifos of the link
423  // have initValues, then the elements that are created are those of the
424  // objFifo with elements of bigger size
425  bool linked = false;
426  auto linkOp = getOptionalLinkOp(op);
427  if (linkOp) {
428  auto fifoIn = linkOp->getInputObjectFifos()[0];
429  auto fifoOut = linkOp->getOutputObjectFifos()[0];
430  linked = true;
431  if (objFifoLinks.find(*linkOp) != objFifoLinks.end())
432  return; // elements have already been created
433  if (linkOp->isJoin()) {
434  // if join, fifoOut has bigger size
435  if (op.name() != fifoOut.name())
436  return;
437  } else if (linkOp->isDistribute()) {
438  // if distribute, fifoIn has bigger size
439  if (op.name() != fifoIn.name())
440  return;
441  } else {
442  // check if output objectfifo has initValues
443  if (fifoOut.getInitValues().has_value()) {
444  if (fifoOut.name() != op.name())
445  return;
446  } else {
447  // check which objectfifo of the link has bigger size
448  auto fifoInType = llvm::cast<AIEObjectFifoType>(fifoIn.getElemType());
449  auto elemInType = llvm::cast<MemRefType>(fifoInType.getElementType());
450  int inSize = elemInType.getNumElements();
451 
452  auto fifoOutType =
453  llvm::cast<AIEObjectFifoType>(fifoOut.getElemType());
454  auto elemOutType =
455  llvm::cast<MemRefType>(fifoOutType.getElementType());
456 
457  if (int outSize = elemOutType.getNumElements(); inSize >= outSize) {
458  if (op.name() != fifoIn.name())
459  return;
460  } else {
461  if (fifoOut.name() != op.name())
462  return;
463  }
464  }
465  }
466  }
467 
468  TileOp creation_tile;
469  if (share_direction == 0 || share_direction == -1)
470  creation_tile = op.getProducerTileOp();
471  else {
472  auto consumerTileOp =
473  dyn_cast<TileOp>(op.getConsumerTiles()[0].getDefiningOp());
474  creation_tile = consumerTileOp;
475  }
476 
477  // Reset opbuilder location to after the last tile declaration
478  Operation *t = nullptr;
479  auto dev = op->getParentOfType<DeviceOp>();
480  for (auto tile_op : dev.getBody()->getOps<TileOp>()) {
481  t = tile_op.getOperation();
482  }
483  builder.setInsertionPointAfter(t);
484  for (int i = 0; i < numElem; i++) {
485  mlir::ElementsAttr initValues = nullptr;
486  if (!creation_tile.isShimTile()) {
487  if (op.getInitValues().has_value()) {
488  initValues =
489  llvm::cast<mlir::ElementsAttr>(op.getInitValues().value()[i]);
490  }
491  auto buff = builder.create<BufferOp>(
492  builder.getUnknownLoc(), elemType, creation_tile,
493  builder.getStringAttr(op.name().str() + "_buff_" +
494  std::to_string(of_elem_index)),
495  /*address*/ nullptr, initValues,
496  /*mem_bank*/ nullptr);
497  buffers.push_back(buff);
498  }
499  of_elem_index++;
500  }
501  int repeatCount = 1;
502  if (op.getRepeatCount().has_value())
503  repeatCount = op.getRepeatCount().value();
504  if (linked) {
505  if (linkOp->getRepeatCount().has_value())
506  repeatCount = linkOp->getRepeatCount().value();
507  if (linkOp->isDistribute())
508  numElem *= linkOp->getFifoOuts().size();
509  else if (linkOp->isJoin())
510  numElem *= linkOp->getFifoIns().size();
511  objFifoLinks[*linkOp] = op;
512  }
513  std::vector<LockOp> locks = createObjectFifoLocks(
514  builder, lockAnalysis, op, numElem, creation_tile, repeatCount);
515  buffersPerFifo[op] = buffers;
516  locksPerFifo[op] = locks;
517  }

555  /// producerDMA (i.e. on consumer tile).

558  int acqNum, int relNum, MyOp buff, int offset, int len,

939  // Function that unrolls for-loops that contain objectFifo operations.
940  LogicalResult unrollForLoops(DeviceOp &device, OpBuilder &builder,
941  std::set<TileOp> objectFifoTiles) {
942  for (auto coreOp : device.getOps<CoreOp>()) {
943  if (objectFifoTiles.count(coreOp.getTileOp()) > 0) {
944  std::vector<scf::ForOp> unrolledLoops;
945  std::map<Operation *, bool> foundMap;
946  std::map<Operation *, int64_t> remainderMap;
947  std::map<Operation *, int64_t> tripCountMap;
948  WalkResult res = coreOp.walk([&](scf::ForOp forLoop) {
949  // look for operations on objectFifos
950  // when multiple fifos in same loop, must use the smallest
951  // common multiplier as the unroll factor
952  foundMap[forLoop.getOperation()] = false;
953  std::set<int> objFifoSizes;
954  Block *body = forLoop.getBody();
955  remainderMap[forLoop.getOperation()] = 0;
956  for (auto acqOp : body->getOps<ObjectFifoAcquireOp>()) {
957  if (acqOp.getOperation()->getParentOp() == forLoop) {
958  foundMap[forLoop.getOperation()] = true;
959  ObjectFifoCreateOp op = acqOp.getObjectFifo();
960  objFifoSizes.insert(op.size());
961  }
962  }
963  // If the loop doesn't have acquire and release locks
964  // Push it to the unrolledLoops to avoid unrolling
965  if (!foundMap[forLoop.getOperation()]) {
966  unrolledLoops.push_back(forLoop);
967  return WalkResult::advance();
968  }
969  // Walk in the loop region to unroll the loop and its remainder
970  Region *region = forLoop->getParentRegion();
971  scf::ForOp prevLoop;
972  prevLoop = forLoop;
973  tripCountMap[prevLoop.getOperation()] = 0;
974  while (remainderMap[prevLoop.getOperation()] > 1 ||
975  foundMap[prevLoop.getOperation()]) {
976  region->walk([&](scf::ForOp remLoop) {
977  bool skipLoop = false;
978  int64_t tripCount = 0;
979  if (remLoop.getSingleLowerBound() &&
980  remLoop.getSingleUpperBound() && remLoop.getSingleStep()) {
981  tripCount = constantTripCount(*(remLoop.getSingleLowerBound()),
982  *(remLoop.getSingleUpperBound()),
983  *(remLoop.getSingleStep()))
984  .value_or(0);
985  }
986  int unrollFactor =
987  computeLCM(objFifoSizes); // also counts original loop body
988  // Loop ids are not unique.
989  // Sometimes, immediately after unrolling, the unrolled loop
990  // and the one next to it (can be the remainder loop or an
991  // independent loop) will have the same ID. This makes it
992  // difficult to identify which loop needs to be unrolled.
993  // Once it restarts walking from start, it ends up allocating
994  // new ID to each loop.
995  if (remainderMap[prevLoop.getOperation()] > 1 &&
996  foundMap[remLoop.getOperation()] == false &&
997  prevLoop != remLoop) {
998  skipLoop = true;
999  }
1000  if (std::count(unrolledLoops.begin(), unrolledLoops.end(),
1001  remLoop) == 0 &&
1002  !skipLoop) {
1003  tripCountMap[remLoop.getOperation()] = tripCount;
1004  // if loop iterations < unrollFactor, unroll the loop fully
1005  if (tripCountMap[remLoop.getOperation()] < unrollFactor)
1006  unrollFactor = tripCountMap[remLoop.getOperation()];
1007  // If unrollFactor = 0,divide by zero
1008  if (unrollFactor == 0) {
1009  remLoop.emitOpError()
1010  << "could not be unrolled with unrollFactor = 0, check "
1011  "loop boundaries."
1012  << "\n";
1013  return WalkResult::interrupt();
1014  }
1015  remainderMap[remLoop.getOperation()] =
1016  tripCountMap[remLoop.getOperation()] % unrollFactor;
1017  auto step = remLoop.getStep()
1018  .getDefiningOp<arith::ConstantOp>()
1019  .getValue();
1020  int64_t step_value = llvm::dyn_cast<IntegerAttr>(step).getInt();
1021 
1022  if (step_value < unrollFactor ||
1023  foundMap[remLoop.getOperation()]) {
1024  // Process the for loop
1025  if (failed(mlir::loopUnrollByFactor(remLoop, unrollFactor))) {
1026  remLoop.emitOpError()
1027  << "could not be unrolled with unrollFactor: "
1028  << unrollFactor << "\n";
1029  return WalkResult::interrupt();
1030  }
1031  unrolledLoops.push_back(remLoop);
1032  foundMap[remLoop.getOperation()] = false;
1033  } else {
1034  remainderMap[remLoop.getOperation()] = 0;
1035  foundMap[remLoop.getOperation()] = false;
1036  }
1037  } else {
1038  remainderMap[remLoop.getOperation()] = 0;
1039  foundMap[remLoop.getOperation()] = false;
1040  }
1041  prevLoop = remLoop;
1042  return WalkResult::advance();
1043  });
1044  }
1045  return WalkResult::advance();
1046  });
1047  if (res.wasInterrupted())
1048  return failure();
1049  }
1050  }
1051  return success();
1052  }

1075  // Function that generates the IR for objectfifo accesses to be handled at
1076  // runtime.
1077  LogicalResult dynamicGlobalObjectFifos(DeviceOp &device, OpBuilder &builder,
1078  std::set<TileOp> objectFifoTiles) {
1079  for (auto coreOp : device.getOps<CoreOp>()) {
1080  if (objectFifoTiles.count(coreOp.getTileOp()) <= 0)
1081  continue;
1082  if (objectFifoTiles.count(coreOp.getTileOp()) > 0) {
1083  // For each core: count the number of objectFifos and create
1084  // a global buffer just before the core to track index of
1085  // next object to access.
1086  // !! NOTE !! objectFifos with same producer / consumer tile
1087  // need two counters (accessed based on the ObjectFifoPort)
1088  std::map<std::pair<ObjectFifoCreateOp, ObjectFifoPort>, int> fifoSizes;
1089  coreOp.walk([&](ObjectFifoAcquireOp acqOp) {
1090  ObjectFifoCreateOp op = acqOp.getObjectFifo();
1091  ObjectFifoPort port = acqOp.getPort();
1092  if (fifoSizes.find({op, port}) == fifoSizes.end())
1093  fifoSizes[{op, port}] = op.size();
1094  });
1095  builder.setInsertionPoint(coreOp);
1096  auto memrefTy =
1097  MemRefType::get(SmallVector<int64_t>{(int64_t)fifoSizes.size()},
1098  builder.getI32Type());
1099  auto globalNextIndex = builder.create<BufferOp>(
1100  builder.getUnknownLoc(), memrefTy, coreOp.getTile(),
1101  /*sym_name*/ nullptr, /*address*/ nullptr,
1102  /*initial_value*/ nullptr, /*mem_bank*/ nullptr);
1103 
1104  // Initialize all counters in the global buffers to 0.
1105  // Also, keep a map of the ConstantOps for the indices per OF
1106  // and a map with the ConstantOps for the sizes per OF.
1107  std::map<std::pair<ObjectFifoCreateOp, ObjectFifoPort>,
1108  arith::ConstantOp>
1109  globalIndices;
1110  std::map<std::pair<ObjectFifoCreateOp, ObjectFifoPort>,
1111  arith::ConstantOp>
1112  constantSizes;
1113  int index = 0;
1114  builder.setInsertionPointToStart(&(coreOp.getBody().front()));
1115  Value initVal = builder.create<arith::ConstantOp>(
1116  builder.getUnknownLoc(), builder.getI32IntegerAttr(0));
1117  for (auto i : fifoSizes) {
1118  auto indexOp = builder.create<arith::ConstantOp>(
1119  initVal.getLoc(), builder.getIndexAttr(index));
1120  globalIndices[i.first] = indexOp;
1121  index++;
1122  auto size = builder.create<arith::ConstantOp>(
1123  indexOp.getLoc(), builder.getI32IntegerAttr(i.second));
1124  constantSizes[i.first] = size;
1125  builder.create<memref::StoreOp>(
1126  size.getLoc(), initVal, globalNextIndex,
1127  ValueRange(ArrayRef({indexOp.getResult()})));
1128  }
1129 
1130  // Walk the code:
1131  // - after each ObjectFifoReleaseOp:
1132  // - globalNextIndex: add #rel modulo objfifo depth
1133  // - before each ObjectFifoAcquireOp:
1134  // - globalNextIndex: load index and use it to index_switch (one
1135  // IndexSwithOp per AccessOp)
1136  WalkResult res = coreOp.walk([&](Operation *op) {
1137  if (auto relOp = dyn_cast<ObjectFifoReleaseOp>(op)) {
1138  ObjectFifoCreateOp createOp = relOp.getObjectFifo();
1139  ObjectFifoPort port = relOp.getPort();
1140  updateGlobalNextIndex(builder, relOp, globalNextIndex,
1141  globalIndices[{createOp, port}],
1142  constantSizes[{createOp, port}]);
1143  }
1144  if (auto acqOp = dyn_cast<ObjectFifoAcquireOp>(op)) {
1145  std::vector<ObjectFifoSubviewAccessOp> accessOps;
1146  for (auto u : acqOp->getUsers())
1147  if (auto accessOp = dyn_cast<ObjectFifoSubviewAccessOp>(u))
1148  accessOps.push_back(accessOp);
1149 
1150  for (auto accessOp : accessOps) {
1151  ObjectFifoCreateOp createOp = acqOp.getObjectFifo();
1152  ObjectFifoPort port = acqOp.getPort();
1153 
1154  // Single switch case
1155  if (fifoSizes[{createOp, port}] == 1)
1156  return WalkResult::advance();
1157 
1158  // Create a switch for each subview access
1159  builder.setInsertionPointAfter(accessOp);
1160  auto switchIndexAsInteger = builder.create<memref::LoadOp>(
1161  builder.getUnknownLoc(), globalNextIndex,
1162  ValueRange(
1163  ArrayRef({globalIndices[{createOp, port}].getResult()})));
1164  auto switchIndex = builder.create<arith::IndexCastOp>(
1165  builder.getUnknownLoc(), builder.getIndexType(),
1166  switchIndexAsInteger);
1167  unsigned caseRegionCounts = fifoSizes[{createOp, port}];
1168  SmallVector<int64_t, 4> caseValues;
1169  for (int i = 0; i < fifoSizes[{createOp, port}]; ++i) {
1170  caseValues.push_back(i);
1171  }

1211  /// Function used to create a UseLockOp based on input parameters.
1212  /// acc is an accumulator map that tracks the indices of the next locks to
1213  /// acquire (or release). Uses op to find index of acc for next lockID.
1214  /// Updates acc.
1215  void createUseLocks(OpBuilder &builder, ObjectFifoCreateOp op,
1216  ObjectFifoPort port,
1217  DenseMap<std::pair<ObjectFifoCreateOp, int>, int> &acc,
1218  int numLocks, LockAction lockAction) {
1219  ObjectFifoCreateOp target = op;
1220  auto portNum = port == ObjectFifoPort::Produce ? 0 : 1;
1221  if (auto linkOp = getOptionalLinkOp(op))
1222  if (objFifoLinks.find(*linkOp) != objFifoLinks.end())
1223  target = objFifoLinks[*linkOp];
1224 
1225  auto dev = op->getParentOfType<DeviceOp>();
1226  if (!dev.getTargetModel().hasProperty(AIETargetModel::UsesSemaphoreLocks)) {
1227 
1228  if (locksPerFifo[target].size() == 0) {
1229  for (int i = 0; i < numLocks; i++) {
1230  int lockID = acc[{op, portNum}];
1231  acc[{op, portNum}] =
1232  (lockID + 1) % op.size(); // update to next objFifo elem
1233  }
1234  return;
1235  }
1236 
1237  int lockMode = 0;
1238  if ((port == ObjectFifoPort::Produce &&
1239  lockAction == LockAction::Release) ||
1240  (port == ObjectFifoPort::Consume &&
1241  lockAction == LockAction::Acquire))
1242  lockMode = 1;
1243  for (int i = 0; i < numLocks; i++) {
1244  int lockID = acc[{op, portNum}];
1245  builder.create<UseLockOp>(builder.getUnknownLoc(),
1246  locksPerFifo[target][lockID], lockAction,
1247  lockMode);
1248  acc[{op, portNum}] =
1249  (lockID + 1) % op.size(); // update to next objFifo elem
1250  }
1251  } else {
1252  if (numLocks == 0)
1253  return;
1254 
1255  if (locksPerFifo[target].size() == 0) {
1256  acc[{op, portNum}] = (acc[{op, portNum}] + numLocks) %
1257  op.size(); // update to next objFifo elem
1258  return;
1259  }
1260 
1261  // search for the correct lock based on the port of the acq/rel
1262  // operation e.g. acq as consumer is the read lock (second)
1263  LockOp lock;
1264  if (lockAction == LockAction::AcquireGreaterEqual) {
1265  if (port == ObjectFifoPort::Produce)
1266  lock = locksPerFifo[target][0];
1267  else
1268  lock = locksPerFifo[target][1];
1269  } else {
1270  if (port == ObjectFifoPort::Produce)
1271  lock = locksPerFifo[target][1];
1272  else
1273  lock = locksPerFifo[target][0];
1274  }
1275  builder.create<UseLockOp>(builder.getUnknownLoc(), lock, lockAction,
1276  numLocks);
1277  acc[{op, portNum}] = (acc[{op, portNum}] + numLocks) %
1278  op.size(); // update to next objFifo elem
1279  }
1280  }
