288  }

559 bool AIE2TargetModel::isLegalTileConnection(int col, int row,
560  WireBundle srcBundle, int srcChan,
561  WireBundle dstBundle,
562  int dstChan) const {
563  // Check Channel Id within the range
564  if (srcChan >= int(getNumSourceSwitchboxConnections(col, row, srcBundle)))
565  return false;
566  if (dstChan >= int(getNumDestSwitchboxConnections(col, row, dstBundle)))
567  return false;
568 
569  // Lambda function to check if a bundle is in a list
570  auto isBundleInList = [](WireBundle bundle,
571  std::initializer_list<WireBundle> bundles) {
572  return std::find(bundles.begin(), bundles.end(), bundle) != bundles.end();
573  };
574 
575  // Memtile
576  if (isMemTile(col, row)) {
577  if (srcBundle == WireBundle::DMA) {
578  if (dstBundle == WireBundle::DMA)
579  return srcChan == dstChan;
580  if (isBundleInList(dstBundle, {WireBundle::Ctrl, WireBundle::South,
581  WireBundle::North}))
582  return true;
583  }
584  if (srcBundle == WireBundle::Ctrl) {
585  if (dstBundle == WireBundle::DMA)
586  return dstChan == 5;
587  if (isBundleInList(dstBundle, {WireBundle::South, WireBundle::North}))
588  return true;
589  }
590  if (isBundleInList(srcBundle, {WireBundle::South, WireBundle::North})) {
591  if (isBundleInList(dstBundle, {WireBundle::DMA, WireBundle::Ctrl}))
592  return true;
593  if (isBundleInList(dstBundle, {WireBundle::South, WireBundle::North}))
594  return srcChan == dstChan;
595  }
596  if (srcBundle == WireBundle::Trace) {
597  if (dstBundle == WireBundle::DMA)
598  return dstChan == 5;
599  if (dstBundle == WireBundle::South)
600  return true;
601  }
602  }
603  // Shimtile
604  else if (isShimNOCorPLTile(col, row)) {
605  if (srcBundle == WireBundle::Ctrl)
606  return dstBundle != WireBundle::Ctrl;
607  if (isBundleInList(srcBundle, {WireBundle::FIFO, WireBundle::South}))
608  return isBundleInList(dstBundle, {WireBundle::Ctrl, WireBundle::FIFO,
609  WireBundle::South, WireBundle::West,
610  WireBundle::North, WireBundle::East});
611  if (isBundleInList(srcBundle,
612  {WireBundle::West, WireBundle::North, WireBundle::East}))
613  return (srcBundle == dstBundle)
614  ? (srcChan == dstChan)
615  : isBundleInList(dstBundle,
616  {WireBundle::Ctrl, WireBundle::FIFO,
617  WireBundle::South, WireBundle::West,
618  WireBundle::North, WireBundle::East});
619  if (srcBundle == WireBundle::Trace) {
620  if (isBundleInList(dstBundle, {WireBundle::FIFO, WireBundle::South}))
621  return true;
622  if (isBundleInList(dstBundle, {WireBundle::West, WireBundle::East}))
623  return dstChan == 0;
624  }
625  }
626  // Coretile
627  else if (isCoreTile(col, row)) {
628  if (isBundleInList(srcBundle,
629  {WireBundle::DMA, WireBundle::FIFO, WireBundle::South,
630  WireBundle::West, WireBundle::North, WireBundle::East}))
631  if (isBundleInList(dstBundle,
632  {WireBundle::Core, WireBundle::DMA, WireBundle::Ctrl,
633  WireBundle::FIFO, WireBundle::South, WireBundle::West,
634  WireBundle::North, WireBundle::East}))
635  return (srcBundle == dstBundle) ? (srcChan == dstChan) : true;
636  if (srcBundle == WireBundle::Core)
637  return dstBundle != WireBundle::Core;
638  if (srcBundle == WireBundle::Ctrl)
639  return dstBundle != WireBundle::Ctrl && dstBundle != WireBundle::DMA;
640  if (srcBundle == WireBundle::Trace) {
641  if (dstBundle == WireBundle::DMA)
642  return dstChan == 0;
643  if (isBundleInList(dstBundle, {WireBundle::FIFO, WireBundle::South}))
644  return true;
645  }
646  }
647  return false;
648 }

650 void AIETargetModel::validate() const {
651  // Every tile in a shimtile row must be a shimtile, and can only be one type
652  // of shim tile.
653  for (int j = 0; j < columns(); j++) {
654  assert(!isMemTile(j, 0) && (isShimPLTile(j, 0) || isShimNOCTile(j, 0)) &&
655  !isCoreTile(j, 0));
656  assert(isShimPLTile(j, 0) ^ isShimNOCTile(j, 0));
657  }
658 
659  // Every tile in a memtile row must be a memtile.
660  for (int i = 1; i < 1 + static_cast<int>(getNumMemTileRows()); i++)
661  for (int j = 0; j < columns(); j++)
662  assert(isMemTile(j, i) && !isShimPLTile(j, i) && !isShimNOCTile(j, i) &&
663  !isCoreTile(j, i));
664 
665  // Every other tile is a coretile.
666  for (int i = 1 + getNumMemTileRows(); i < rows(); i++)
667  for (int j = 0; j < columns(); j++)
668  assert(!isMemTile(j, i) && !isShimPLTile(j, i) && !isShimNOCTile(j, i) &&
669  isCoreTile(j, i));
670 
671  // Looking North, buses must match
672  for (int i = 0; i < rows() - 1; i++)
673  for (int j = 0; j < columns(); j++)
674  assert(getNumSourceSwitchboxConnections(j, i, WireBundle::North) ==
675  getNumDestSwitchboxConnections(j, i + 1, WireBundle::South));
676  // Looking South, buses must match
677  for (int i = 1; i < rows(); i++)
678  for (int j = 0; j < columns(); j++)
679  assert(getNumSourceSwitchboxConnections(j, i, WireBundle::South) ==
680  getNumDestSwitchboxConnections(j, i - 1, WireBundle::North));
681  // Looking East, buses must match
682  for (int i = 0; i < rows(); i++)
683  for (int j = 0; j < columns() - 1; j++)
684  assert(getNumSourceSwitchboxConnections(j, i, WireBundle::East) ==
685  getNumDestSwitchboxConnections(j + 1, i, WireBundle::West));
686  // Looking West, buses must match
687  for (int i = 0; i < rows(); i++)
688  for (int j = 1; j < columns(); j++)
689  assert(getNumSourceSwitchboxConnections(j, i, WireBundle::West) ==
690  getNumDestSwitchboxConnections(j - 1, i, WireBundle::East));
691  // Edges have no connections
692  for (int j = 0; j < columns(); j++)
693  assert(getNumSourceSwitchboxConnections(j, rows() - 1, WireBundle::North) ==
694  0);
695  for (int i = 0; i < rows(); i++)
696  assert(getNumSourceSwitchboxConnections(columns() - 1, i,
697  WireBundle::East) == 0);
698  for (int i = 0; i < rows(); i++)
699  assert(getNumSourceSwitchboxConnections(0, i, WireBundle::West) == 0);
700 
701  // FIFOS are consistent
702  for (int i = 0; i < rows(); i++)
