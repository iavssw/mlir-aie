31 
32  DMAStartBdChainForOpPattern(MLIRContext *ctx)
33  : RewritePattern(DMAStartBdChainForOp::getOperationName(),
34  PatternBenefit(1), ctx) {}
35 
36  LogicalResult matchAndRewrite(Operation *op_any,
37  PatternRewriter &rewriter) const override {
38  DMAStartBdChainForOp op = llvm::dyn_cast<DMAStartBdChainForOp>(op_any);
39  if (!op) {
40  return failure();
41  }
42  AIE::DeviceOp device = op->getParentOfType<AIE::DeviceOp>();
43 
44  AIE::ShimDMAAllocationOp alloc_op =
45  AIE::ShimDMAAllocationOp::getForSymbol(device, op.getAlloc());
46  if (!alloc_op) {
47  return op.emitOpError("no shim DMA allocation found for symbol");
48  }
49 
50  const int col = alloc_op.getCol();
51  AIE::TileOp tile = AIE::TileOp::getOrCreate(rewriter, device, col, 0);
52  DMAStartBdChainOp new_op = rewriter.create<DMAStartBdChainOp>(
53  op.getLoc(), rewriter.getIndexType(), op.getSymbol(), op.getArgs(),
54  tile.getResult(), alloc_op.getChannelDir(),
55  (int32_t)alloc_op.getChannelIndex(), op.getIssueToken(),
56  op.getRepeatCount());
57  rewriter.replaceAllUsesWith(op.getResult(), new_op.getResult());
58  rewriter.eraseOp(op);
59  return success();
60  }

63 struct DMAInlineBDChainPattern : RewritePattern {
64 
65  DMAInlineBDChainPattern(MLIRContext *ctx)
66  : RewritePattern(DMAStartBdChainOp::getOperationName(), PatternBenefit(1),
67  ctx) {}
68 
69  LogicalResult matchAndRewrite(Operation *op,
70  PatternRewriter &rewriter) const override {
71  DMAStartBdChainOp start_op = llvm::dyn_cast<DMAStartBdChainOp>(op);
72  if (!start_op) { // Not a match.
73  return failure();
74  }
75  rewriter.setInsertionPointAfter(start_op);
76 
77  // Get referenced abstract BD chain
78  AIE::BDChainOp chain_def = start_op.getBDChainOp();
79  assert(chain_def);
80  Region &source_region = chain_def.getBody();
81 
82  // Create BD op into which the result will be inlined
83  DMAConfigureTaskOp configure_op = rewriter.create<DMAConfigureTaskOp>(
84  start_op.getLoc(), rewriter.getIndexType(), start_op.getTile(),
85  start_op.getDirection(), start_op.getChannel(),
86  start_op.getIssueToken(), start_op.getRepeatCount());
87  Region &target_region = configure_op.getBody();
88 
89  // Clone BD definition into usage site, replacing abstract SSA values with
90  // concrete ones
91  IRMapping arg_map;
92  ValueRange values = start_op.getArgs();
93  for (unsigned i = 0, n = source_region.getNumArguments(); i < n; i++) {
94  BlockArgument arg = source_region.getArgument(i);
95  Value val = values[i];
96  assert(arg.getType() == val.getType());
97  arg_map.map(arg, val);
98  }
99  source_region.cloneInto(&target_region, arg_map);
100 
101  // Replace result of dma start task with result of bd chain configuration
102  rewriter.replaceAllUsesWith(start_op.getResult(), configure_op.getResult());
103 
104  // Add a start BDs instruction
105  rewriter.create<DMAStartTaskOp>(start_op.getLoc(),
106  configure_op.getResult());
107 
108  // After fully inlining, remove the original instruction
109  rewriter.eraseOp(start_op);
110 
111  return success();
112  }
113 };
